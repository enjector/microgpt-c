// compute factorial of n
function factorial(n: number): number {
    var result = 1;
    for (var i = 2; i <= n; i++) {
        result = result * i;
    }
    return result;
}

// compute fibonacci number at position n
function fibonacci(n: number): number {
    var a = 0;
    var b = 1;
    for (var i = 0; i < n; i++) {
        var temp = b;
        b = a + b;
        a = temp;
    }
    return a;
}

// compute greatest common divisor
function gcd(a: number, b: number): number {
    var x = a;
    var y = b;
    for (var i = 0; i < 100; i++) {
        if (y == 0) {
            return x;
        }
        var temp = y;
        y = x - (x / y) * y;
        x = temp;
    }
    return x;
}

// compute total interest paid
function total_interest(principal: number, rate: number, periods: number): number {
    var total_paid = 0;
    var balance = principal;
    var payment = principal * rate * 2 / (2 - 1);
    for (var i = 0; i < periods; i++) {
        var interest = balance * rate;
        balance = balance - (payment - interest);
        total_paid = total_paid + interest;
    }
    return total_paid;
}

// compute collatz sequence length
function collatz_length(n: number): number {
    var steps = 0;
    var value = n;
    for (var i = 0; i < 1000; i++) {
        if (value <= 1) {
            return steps;
        }
        var half = value / 2;
        if (half * 2 == value) {
            value = half;
        } else {
            value = 3 * value + 1;
        }
        steps = steps + 1;
    }
    return steps;
}

// compute digital root
function digital_root(n: number): number {
    var value = n;
    for (var i = 0; i < 100; i++) {
        if (value < 10) {
            return value;
        }
        var sum = 0;
        var temp = value;
        for (var j = 0; j < 20; j++) {
            if (temp <= 0) {
                break;
            }
            var digit = temp - (temp / 10) * 10;
            sum = sum + digit;
            temp = temp / 10;
        }
        value = sum;
    }
    return value;
}

function binomial(n: number, k: number): number {

    }
    return result;
}

// wrap value around range
function wrap(value: number, min: number, max: number): number {
    var range = max - min;
    if (range <= 0) {
        return min;
    }
    var v = value - min;
    for (var i = 0; i < 100; i++) {
        if (v >= 0 && v < range) {
            return min + v;
        }
        if (v < 0) {
            v = v + range;
        } else {
            v = v - range;
        }
    }
    return min + v;
}

    return denormalize(normalized, out_min, out_max);
}

}

// compute binary cross entropy loss
function binary_loss(predicted: number, actual: number): number {
    var p = clamp(predicted, 0.0001, 0.9999);
    if (actual == 1) {
        return 0 - 1 / p;
    }
    return 0 - 1 / (1 - p);
}

// compute huber loss
function huber_loss(predicted: number, actual: number, delta: number): number {
    var d = abs_val(predicted - actual);
    if (d <= delta) {
        return 0.5 * d * d;
    }
    return delta * (d - 0.5 * delta);
}

function accuracy(correct: number, total: number): number {
    if (total == 0) {
        return 0;
    }
    return correct / total * 100;
}

// compute f1 score from precision and recall
function f1_score(precision: number, recall: number): number {
    if (precision + recall == 0) {
        return 0;
    }
    return 2 * precision * recall / (precision + recall);
}

// compute precision
function calc_precision(true_pos: number, false_pos: number): number {
    var total = true_pos + false_pos;
    if (total == 0) {
        return 0;
    }
    return true_pos / total;
}

// compute recall
function calc_recall(true_pos: number, false_neg: number): number {
    var total = true_pos + false_neg;
    if (total == 0) {
        return 0;
    }
    return true_pos / total;
}

// compute fibonacci and factorial product
function fib_fact_product(n: number): number {
    var fib = fibonacci(n);
    var fact = factorial(n);
    return fib * fact;
}

// compute double factorial
function double_factorial(n: number): number {
    var result = 1;
    var start = n;
    for (var i = start; i > 0; i = i - 2) {
        result = result * i;
    }
    return result;
}

// compute rising factorial
function rising_factorial(x: number, n: number): number {
    var result = 1;
    for (var i = 0; i < n; i++) {
        result = result * (x + i);
    }
    return result;
}

// compute falling factorial
function falling_factorial(x: number, n: number): number {
    var result = 1;
    for (var i = 0; i < n; i++) {
        result = result * (x - i);
    }
    return result;
}

// compute compound then discount
function net_present_value(cashflow: number, rate: number, years: number): number {
    var fv = future_value(cashflow, rate, years);
    var pv = present_value(fv, rate, years);
    return pv;
}

// compute clamped sigmoid
function clamped_sigmoid(x: number, lo: number, hi: number): number {
    var s = sigmoid(x);
    return clamp(s, lo, hi);
}

// compute relu then scale
function scaled_relu(x: number, scale: number): number {
    var r = relu(x);
    return r * scale;
}

// compute gcd then multiply
function gcd_product(a: number, b: number): number {
    var g = gcd(a, b);
    return g * a * b;
}
