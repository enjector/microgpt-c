function digital_root(n: number): number {
    var value = n;
    for (var i = 0; i < 100; i++) {
        if (value < 10) {
            return value;
        }
        var sum = 0;
        var temp = value;
        for (var j = 0; j < 20; j++) {
            if (temp <= 0) {
                break;
            }
            var digit = temp - (temp / 10) * 10;
            sum = sum + digit;
            temp = temp / 10;
        }
        value = sum;
    }
    return value;
}

// calculate geometric sum
function geometric_sum(ratio: number, terms: number): number {
    var total = 0;
    var power = 1;
    for (var i = 0; i < terms; i++) {
        total = total + power;
