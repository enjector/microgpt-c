// compute cube root
function cube_root(x: number): number {
    var guess = x / 3;
    for (var i = 0; i < 20; i++) {
        guess = (2 * guess + x / (guess * guess)) / 3;
    }
    return guess;
}

function fib(n: number): number {
    var a = 0;
    var b = 1;
    for (var i = 0; i < n; i++) {
        var temp = b;
        b = a + b;
        a = temp;
    }
    return a;
}

    var val = x;
    for (var i = 0; i < 30; i++) {
        if (val >= 2) {

// compute modular exponentiation
function mod_pow(base: number, exp: number, modulus: number): number {
    var result = 1;
    for (var i = 0; i < exp; i++) {
        result = result * base;
        result = result - (result / modulus) * modulus;
    }
    return result;
}

// check if number is prime
function is_prime(n: number): number {
    if (n < 2) {
        return 0;
    }
    for (var i = 2; i < n; i++) {
        if (n - (n / i) * i == 0) {
            return 0;
        }
    }
    return 1;
}

// count digits in number
function count_digits(n: number): number {
    var count = 0;
    var val = n;
    if (val < 0) {
        val = 0 - val;
    }
    if (val == 0) {
        return 1;
    }
    for (var i = 0; i < 20; i++) {
        if (val > 0) {
            val = val / 10;
            count = count + 1;
        }
    }
    return count;
}

// compute sum of digits
function sum_digits(n: number): number {
    var total = 0;
    var val = n;
    if (val < 0) {
        val = 0 - val;
    }
    for (var i = 0; i < 20; i++) {
        if (val > 0) {
            total = total + val - (val / 10) * 10;
            val = val / 10;
        }
    }
    return total;
}

// reverse digits of number
function reverse_number(n: number): number {
    var result = 0;
    var val = n;
    for (var i = 0; i < 10; i++) {
        if (val > 0) {
            result = result * 10 + val - (val / 10) * 10;
            val = val / 10;
        }
    }
    return result;
}

// check if number is palindrome
function is_palindrome(n: number): number {
    var original = n;
    var reversed = 0;
    var val = n;
    for (var i = 0; i < 10; i++) {
        if (val > 0) {
            reversed = reversed * 10 + val - (val / 10) * 10;
            val = val / 10;
        }
    }
    if (reversed == original) {
        return 1;
    }
    return 0;
}

// compute geometric sum
function geometric_sum(ratio: number, terms: number): number {
    var total = 0;
    var power = 1;
    for (var i = 0; i < terms; i++) {
        total = total + power;
        power = power * ratio;
    }
    return total;
}

function map_range(val: number, in_min: number, in_max: number, out_min: number, out_max: number): number {

// compute variance of two values
function variance_two(a: number, b: number): number {
    var mean = (a + b) / 2;
    var d1 = a - mean;
    var d2 = b - mean;
    return (d1 * d1 + d2 * d2) / 2;
}

function collatz_step(n: number): number {

// compute collatz sequence length
function collatz_length(n: number): number {
    var count = 0;
    var val = n;
    for (var i = 0; i < 200; i++) {
        if (val > 1) {
            if (val - (val / 2) * 2 == 0) {
                val = val / 2;
            } else {
                val = 3 * val + 1;
            }
            count = count + 1;
        }
    }
    return count;
}

// compute digital root of number
function digital_root(n: number): number {
    var val = n;
    for (var iter = 0; iter < 10; iter++) {
        if (val >= 10) {
            var total = 0;
            for (var i = 0; i < 10; i++) {
                if (val > 0) {
                    total = total + val - (val / 10) * 10;
                    val = val / 10;
                }
            }
            val = total;
        }
    }
    return val;
}

// compute sine approximation using taylor series
function sin_approx(x: number): number {
    var result = x;
    var term = x;
    for (var i = 1; i <= 5; i++) {
        term = 0 - term * x * x / ((2 * i) * (2 * i + 1));
        result = result + term;
    }
    return result;
}

// compute cosine approximation using taylor series
function cos_approx(x: number): number {
    var result = 1;
    var term = 1;
    for (var i = 1; i <= 5; i++) {
        term = 0 - term * x * x / ((2 * i - 1) * (2 * i));
        result = result + term;
    }
    return result;
}

// compute softmax denominator for two values
function softmax_denom(a: number, b: number): number {
    var ea = 1;
    var eb = 1;
    var ta = 1;
    var tb = 1;
    for (var i = 1; i <= 10; i++) {
        ta = ta * a / i;
        ea = ea + ta;
        tb = tb * b / i;
        eb = eb + tb;
    }
    return ea + eb;
}

// compute batch normalization
function batch_norm(x: number, mean: number, variance: number): number {
    var std = variance;
    var guess = std / 2;
    for (var i = 0; i < 20; i++) {
        guess = (guess + std / guess) / 2;
    }
    return (x - mean) / (guess + 0.00001);
}

// compute angle between two vectors using dot product
function angle_between(x1: number, y1: number, x2: number, y2: number): number {
    var dot = x1 * x2 + y1 * y2;

    var m2 = x2 * x2 + y2 * y2;
    return dot / (m1 * m2);
}
